\begin{abstract}\noindent
Mô hình \emph{hợp nhất}: mọi kiểm toán là một thể hiện của bất đẳng thức LRFD
$\sum\eta\gamma Q \le \phi R_n$ (TCVN 11823:2017 --- bản Việt hóa của AASHTO LRFD).
Không có ràng buộc ``bật/tắt'': tất cả trạng thái giới hạn (Cường độ, Sử dụng, Đặc
biệt) luôn áp dụng. Trị số $\phi,\gamma$ và phương pháp lấy theo TCVN 11823 phần 3
(tải), phần 5 (bê tông), phần 10 (nền móng); các trị số trong tài liệu là giá trị
AASHTO LRFD mà TCVN 11823 áp dụng (cần đối chiếu bản chuẩn khi xuất hồ sơ).
\end{abstract}

% =========================================================================
\section{Nguyên lý hợp nhất (LRFD)}
% =========================================================================
Mọi cấu kiện, mọi trạng thái giới hạn thỏa cùng một dạng (TCVN 11823-1/-3):
\begin{equation}
\sum_i \eta_i\,\gamma_i\,Q_i \ \le\ \phi\,R_n \;=\; R_r,
\label{eq:lrfd}
\end{equation}
với $\eta=\eta_D\eta_R\eta_I\ge0.95$ (hệ số dẻo--dư thừa--tầm quan trọng), $\gamma_i$
hệ số tải, $Q_i$ hiệu ứng tải, $\phi$ hệ số sức kháng, $R_n$ sức kháng danh định.

\paragraph{Tổ hợp tải (TCVN 11823-3, Bảng 3.4.1-1).}
\begin{center}
\begin{tabular}{@{}lcccl@{}}
\toprule
Trạng thái giới hạn & $\gamma_{DC}$ & $\gamma_{DW}$ & $\gamma_{LL}$ & Ghi chú \\
\midrule
Cường độ I (Strength I) & $1.25/0.90$ & $1.50/0.65$ & $1.75$ & cơ bản có hoạt tải \\
Sử dụng I (Service I)   & $1.00$ & $1.00$ & $1.00$ & lún, chuyển vị \\
Đặc biệt I (Extreme I)  & $1.25/0.90$ & $1.50/0.65$ & $\gamma_{EQ}{=}0.50$ & động đất ($+1.0\,EQ$) \\

\end{tabular}
\end{center}
$\Lset=\{$Strength I, Service I, Extreme I$\}$ là tập tổ hợp tải đã nhân hệ số.

\section{Biến quyết định}
% =========================================================================
\begin{equation}
\mathbf{x}=\big(\,\underbrace{\{\mathbf{p}_i\}}_{\text{bố trí}},\ D,\ L_c,\ (B_x,B_y,h),\ \mathbf{A}_s\,\big),
\end{equation}
bố trí cọc tham số hóa bằng lưới đối xứng $(t,n_x,n_y,s_x,s_y)$, $t\in\{\mathrm A,\mathrm B\}$;
$D\in\Dset$ đường kính; $L_c$ chiều dài hạ cọc; $(B_x,B_y,h)$ đài; $\mathbf{A}_s$ cốt thép đài.
Lưới đối xứng là \emph{cấu tạo bắt buộc} của đài cọc cầu (phân bố lực đều, thi công được).

% =========================================================================
\section{Phương trình trạng thái (vật lý)}
% =========================================================================
Với mỗi tổ hợp $\ell\in\Lset$:
\begin{equation}
\{P_i^\ell,H_i^\ell,M_i^\ell\}=\Anl(\mathbf{x},\Sigma_{\text{soil}};\ell),\qquad
S^\ell=\Set(\mathbf{x},\Sigma_{\text{soil}};\ell),\qquad
\delta^\ell=\text{(phân tích }p\text{--}y).
\end{equation}
$\Anl$ (nội lực cọc) hiện thực bằng \textbf{MCOC} (chính xác). Hiệu ứng nhóm
($p$-multiplier, TCVN 11823-10) luôn nằm trong $\Anl$/$R_n$.

% =========================================================================
\section{Hàm mục tiêu --- chi phí công trình}
% =========================================================================
\begin{equation}
\min_{\mathbf{x}\in\Feas}\ C(\mathbf{x})
= \underbrace{n\frac{\pi D^2}{4}L_c\,c_{bt}+n\,c_{tc}}_{\text{cọc}}
+ \underbrace{(B_xB_yh)\,c_{bt}+W_s(\mathbf{A}_s)\,c_{s}}_{\text{đài}},
\end{equation}
$n=|\{\mathbf{p}_i\}|$ là số cọc --- thành phần chi phối chi phí.

% =========================================================================
\section{Ràng buộc --- mọi trạng thái giới hạn luôn áp dụng}
% =========================================================================
Mỗi ràng buộc viết dạng chuẩn LRFD $\dfrac{\sum\eta\gamma Q}{\phi R_n}-1\le0$.

\paragraph{(A) Cường độ --- cọc} (TCVN 11823-10; $\forall i$, $\forall\ell\in$Strength,Extreme):
\begin{align}
\text{(C1) nén:}\quad & P_{u,i}^\ell \le \phi_c R_n, &
\text{(C2) nhổ:}\quad & T_{u,i}^\ell \le \phi_{up}R_{n,up},\\
\text{(C3) ngang:}\quad & H_{u,i}^\ell \le \phi_\ell R_\ell, &
\text{(C4) tiết diện cọc:}\quad & (P,M)_{u,i}^\ell \in \text{bao } \phi M_n\text{--}\phi P_n.
\end{align}

\paragraph{(B) Cường độ --- đài} (TCVN 11823-5; $\forall\ell\in$Strength):
\begin{align}
\text{(C5) uốn:}\quad & M_u \le \phi_f M_n,\quad M_n=A_sf_y\!\big(d_s-\tfrac a2\big),\ a=\tfrac{A_sf_y}{0.85f'_c b},\\
\text{(C6) cắt 1 phương:}\quad & V_u \le \phi_v V_n,\quad V_n=0.083\,\beta\sqrt{f'_c}\,b_v d_v\ \text{(MPa)},\\
\text{(C7) cắt 2 phương:}\quad & V_u \le \phi_v V_n,\quad V_n=\Big(0.17+\tfrac{0.33}{\beta_c}\Big)\sqrt{f'_c}\,b_o d_v \le 0.33\sqrt{f'_c}\,b_o d_v,\\
\text{(C8) giàn ảo (đài sâu):}\quad & \text{chống/nút/kéo} \le \phi_{stm}\,(\,\text{sức kháng}\,).
\end{align}

\paragraph{(C) Sử dụng} (TCVN 11823-10; $\forall\ell\in$Service I):
\begin{equation}
\text{(C9) lún: } S^\ell \le S_{tol};\qquad
\text{(C10) chuyển vị ngang: } \delta^\ell \le \delta_{tol}.
\end{equation}

\paragraph{(D) Cấu tạo \& thi công} (TCVN 11823-10, $\forall i\neq j$):
\begin{align}
\text{(C11) khoảng cách tim:}\ & \norm{\mathbf{p}_i-\mathbf{p}_j}\ge s_{\min}=\max(2.5D,\,750\,\text{mm}),\\
\text{(C12) cọc cách mép đài:}\ & \ge \tfrac{D}{2}+225\,\text{mm},\qquad
\text{(C13) cọc trong đài.}
\end{align}

% =========================================================================
\section{Phát biểu hợp nhất (gọn)}
% =========================================================================
\begin{equation}
\boxed{
\begin{aligned}
\min_{\mathbf{x}}\ & C(\mathbf{x})=C_{\text{cọc}}+C_{\text{đài}}\\
\text{s.t.}\ & \{P,H,M\}_i^\ell=\Anl(\mathbf{x},\Sigma;\ell),\ S^\ell=\Set(\cdot),\ \delta^\ell && \forall\ell\in\Lset\\
& \text{Cường độ: C1--C8} && \forall i,\ \forall\ell\in\{\text{Str,Ext}\}\\
& \text{Sử dụng: C9--C10} && \forall\ell\in\text{Ser}\\
& \text{Cấu tạo: C11--C13} && \forall i\neq j\\
& \text{mọi ràng buộc dạng } \textstyle\sum\eta\gamma Q\le\phi R_n && \text{(TCVN 11823)}\\
& \mathbf{x}:\ n\in\Integer_+,\ D\in\Dset,\ L_c,B_x,B_y,h\in\Real_+,\ \text{lưới đối xứng}
\end{aligned}}
\end{equation}

\section{Tính giải được \& thuật toán}
% =========================================================================
Bài toán là MINLP, đa tổ hợp tải, ràng buộc tính qua hộp đen ($\Anl$=MCOC; $\Set$;
$p$--$y$) --- không khả vi, không lồi. Mọi ràng buộc C1--C13 đã ở dạng vô thứ nguyên
$\sum\eta\gamma Q/(\phi R_n)-1\le0$ nên gộp được thành \emph{một} độ đo vi phạm:
\begin{equation}
\Theta(\mathbf{x})=\sum_{c}\pos{\frac{\sum\eta\gamma Q_c}{\phi_c R_{n,c}}-1}\ \ge0,
\qquad \text{khả thi}\iff\Theta=0,\qquad \Feas=\{\mathbf{x}:\Theta=0\}.
\end{equation}
Giải bằng \textbf{NSGA-II}: trội có ràng buộc (khả thi thắng bất khả thi; ít vi phạm
thắng; rồi cực tiểu $C$), gieo hạt tất định, cache theo spec lưới, trần ngân sách gọi
$\Anl$. Pseudocode: \texttt{docs/THUAT\_TOAN\_PSEUDOCODE.md}; kế hoạch chuyển đổi:
\texttt{plans/025-migrate-tcvn11823.md}.

\paragraph{Nguồn trị số.} $\gamma$ (Bảng 3.4.1-1), $\phi$ cọc (Đ.10.5.5.2.3), $\phi$
bê tông (Đ.5.5.4.2), $s_{\min}$ \& mép (Đ.10.7.1.2), $p$-multiplier (Đ.10.7.2.4) ---
theo AASHTO LRFD mà TCVN 11823:2017 áp dụng; đối chiếu bản TCVN khi phát hành hồ sơ.
