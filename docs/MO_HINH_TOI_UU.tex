\begin{abstract}\noindent
Tài liệu phát biểu \emph{bài toán thực} (Phần I) một cách đầy đủ, độc lập với cách
giải; sau đó nêu các \emph{giả định kỹ thuật} dùng để thu giảm bài toán (Phần II)
thành một mô hình \emph{giải được} (Phần III); cuối cùng trình bày \emph{thuật toán}
giải (Phần IV). Phần I là chuẩn mực để đánh giá tính đúng đắn; các phần sau là lựa
chọn kỹ thuật để hiện thực.
\end{abstract}

% =========================================================================
\section{PHẦN I --- Mô hình tổng quát (bài toán thực)}
% =========================================================================

\subsection{Phát biểu lời}
Thiết kế nhóm cọc đỡ một trụ cầu sao cho \textbf{tổng chi phí công trình nhỏ nhất}
mà \textbf{thỏa mọi yêu cầu an toàn và cấu tạo theo tiêu chuẩn}, dưới \textbf{mọi tổ
hợp tải trọng} xét đến.

\subsection{Dữ liệu cho trước}
\begin{tabular}{@{}ll@{}}
\toprule
Ký hiệu & Ý nghĩa \\
\midrule
$\{\ell_k\}_{k=1}^{K}$ & tập tổ hợp tải tại đỉnh đài; $\ell_k=(N_k,H^x_k,H^y_k,M^x_k,M^y_k,M^z_k)$ \\
$\Sigma_{\text{soil}}$ & hồ sơ địa chất theo độ sâu: lớp $(h_i,E_i,\varphi_i,c_i,\gamma_i,I_{L,i})$ \\
$(R_b,R_{bt},E_b),(R_s,E_s)$ & cường độ/mô đun bê tông và cốt thép \\
$\mathcal{R}$ & vùng mặt bằng cho phép đặt đài \\
TCVN & 10304:2014, 9362:2012, 5574:2018 \\
\bottomrule
\end{tabular}

\subsection{Biến quyết định (đầy đủ)}
\begin{equation}
n\in\Integer_{+},\quad
\mathbf{p}_i=(x_i,y_i)\in\Real^2,\quad
d_i\in\Dset,\quad
L_{c,i}\in\Real_{+},\quad
(L_x,L_y,H_\partial)\in\Real_{+}^3,\quad
\mathbf{A}_s,
\end{equation}
với $i=1,\dots,n$: số cọc, vị trí tim cọc (\emph{tự do}, không giả định lưới), đường
kính, chiều dài hạ cọc, kích thước--chiều cao đài, và bố trí cốt thép đài $\mathbf{A}_s$.
Đây là bài toán \emph{hỗn hợp nguyên--phi tuyến, số chiều biến thiên theo} $n$.

\subsection{Phương trình trạng thái (vật lý)}
Nội lực cọc bị chi phối bởi cân bằng + tương thích của hệ đài--cọc--đất; với mỗi tổ hợp $k$:
\begin{equation}
\big\{P_i^{k},\,H_i^{k},\,M_i^{k}\big\}_{i=1}^{n}
=\Anl\!\big(\{\mathbf{p}_i,d_i,L_{c,i}\},\,\Sigma_{\text{soil}};\,\ell_k\big),
\label{eq:state}
\end{equation}
trong đó $\Anl$ là \emph{toán tử phân tích kết cấu} (giải hệ đài cứng / tương tác đất--cọc).
$P_i^k$ dương quy ước là nén.

\subsection{Hàm mục tiêu --- chi phí tổng}
\begin{equation}
\min\ C
= \underbrace{\sum_{i=1}^{n}\frac{\pi d_i^{2}}{4}\,L_{c,i}\,c_{bt}+ n\,c_{tc}}_{C_{\text{cọc}}}
+ \underbrace{(L_x L_y H_\partial)\,c_{bt}+ W_s(\mathbf{A}_s)\,c_{s}}_{C_{\text{đài}}},
\end{equation}
với $c_{bt},c_s$ đơn giá bê tông/thép, $c_{tc}$ chi phí thi công mỗi cọc, $W_s$ khối
lượng thép. (Có thể mở rộng \emph{đa mục tiêu}: chi phí vs.\ dự trữ an toàn / độ lún.)

\subsection{Ràng buộc}
\paragraph{(a) Cường độ --- ULS} ($\forall i,\ \forall k$):
\begin{align}
P_i^{k} &\le \Rcd, & \text{(nén)}\\
P_i^{k} &\ge -\Rtd, & \text{(nhổ)}\\
H_i^{k} &\le [H], & \text{(lực ngang)}\\
\frac{P_i^{k}}{\Rcd}+\frac{M_i^{k}}{[M]} &\le 1, & \text{(tương tác P--M)}\\
\text{tiết diện cọc đủ chịu } &(P_i^k,M_i^k), &
\end{align}
với sức chịu tải thiết kế $\Rcd=\dfrac{\gamma_0}{\gamma_n}\dfrac{R_{c,k}}{\gamma_k}$ (TCVN 10304:2014, Đ.7.1.11), bản thân là hàm của đất và số cọc.

\paragraph{(b) Sử dụng --- SLS}
\begin{align}
S &\le S_{gh}, & \text{(độ lún, TCVN 7.4 + Phụ lục C)}\\
\text{nền dưới móng khối quy ước} &\text{ đủ sức chịu}, & \text{(TCVN 7.4)}
\end{align}
với $S=S_e+S_{\text{khối}}$ tính theo móng khối quy ước, ứng suất Boussinesq.

\paragraph{(c) Hình học \& thi công} ($\forall i\neq j$):
\begin{align}
\norm{\mathbf{p}_i-\mathbf{p}_j} &\ge s_{\min}=\max(3d_i,\,d_i+\delta_{\text{tt}}), & \text{(khoảng cách min, TCVN)}\\
\mathbf{p}_i &\in \mathcal{R}_{\ominus c}, & \text{(cọc trong đài, cách mép } c)\\
\norm{\mathbf{p}_i-\mathbf{p}_j} &\le s_{\max} \ \text{(tùy chọn)}, & \text{(giữ giả thiết đài cứng)}
\end{align}

\paragraph{(d) Kết cấu đài} (TCVN 5574:2018): tồn tại $H_\partial,\mathbf{A}_s$ sao cho đài đủ khả năng chịu \emph{uốn}, \emph{chọc thủng} (quanh cột và quanh cọc), \emph{cắt một phương}.

\paragraph{(e) Tuân thủ tiêu chuẩn:} mọi giá trị giới hạn lấy theo TCVN tương ứng.

\subsection{Dạng tổng quát (gọn)}
\begin{equation}
\boxed{
\begin{aligned}
\min_{\substack{n,\{\mathbf{p}_i,d_i,L_{c,i}\}\\ L_x,L_y,H_\partial,\mathbf{A}_s}}\quad
& C\\
\text{s.t.}\quad
& \text{(\ref{eq:state}) phương trình trạng thái} && \forall k\\
& \text{(a) ULS} && \forall i,\forall k\\
& \text{(b) SLS: lún, nền khối}\\
& \text{(c) hình học \& thi công} && \forall i\neq j\\
& \text{(d) kết cấu đài}\\
& n\in\Integer_{+},\ \mathbf{p}_i\in\Real^2,\ d_i\in\Dset,\ L_{c,i},L_x,L_y,H_\partial\in\Real_{+}
\end{aligned}}
\end{equation}
$\Rightarrow$ quy hoạch \emph{hỗn hợp nguyên--phi tuyến, đa tổ hợp tải, số chiều biến thiên},
hàm ràng buộc tính qua \emph{hộp đen} $\Anl$ (không khả vi, không lồi).

\section{PHẦN II --- Giả định và thu giảm}

Bài toán Phần I là bất khả thi về tính toán nếu giữ vị trí cọc tự do. Ta thu giảm bằng các giả định kỹ thuật hợp lý:

\section{PHẦN III --- Mô hình rút gọn (giải được)}
% =========================================================================
\subsection{Biến quyết định}
$\mathbf{x}=(t,n_x,n_y,s_x,s_y)$, với $t\in\{\mathrm{A},\mathrm{B}\}$;
$n_x,n_y\in\Integer_{+}$; $s_x,s_y\in[s_{\min},6d]$. Trong vòng lõi: $L_x,L_y,d$ \textbf{cố định}.

\subsection{Toán tử}
Sinh tọa độ $\mathcal{C}=\Gen(\mathbf{x})$ (lưới đối xứng quanh tâm bệ; kiểu B hàng lẻ
giảm một cọc, tự lệch $s_x/2$). Nội lực $\{P_i^k\}=\Mcoc(\mathcal{C};\{\ell_k\})$;
$P_{\max}=\max_{i,k}P_i^k$, $P_{\min}=\min_{i,k}P_i^k$; $n=|\mathcal{C}|$.

\subsection{Hàm mục tiêu (đa mục tiêu, KHÔNG phạt)}
\begin{equation}
\min_{\mathbf{x}\in\Feas}\ \big(f_1,f_2\big),\qquad
f_1=n,\qquad
f_2=\begin{cases}\Phi(\mathcal{C}) & \text{``bệ gọn''}\\ P_{\max} & \text{``an toàn''}\end{cases}
\end{equation}
Nghiệm là mặt Pareto; phương án kiến nghị $=\arg\min_{\mathbf{x}\in\Feas}(n,f_2)$ theo thứ tự từ điển.

\subsection{Ràng buộc cứng (lõi $=$ R1--R6)}
\begin{equation}
\begin{aligned}
\text{(H1)}&\ n\ge1 &
\text{(H2)}&\ s^{\star}\ge s_{\min}=\max(3d,d+\delta_{\text{tt}})\\
\text{(H3,H4)}&\ \max_i|x_i|+c\le \tfrac{L_x}{2},\ \max_i|y_i|+c\le\tfrac{L_y}{2} &
\text{(H5)}&\ P_{\max}\le[P_o]\\
\text{(H6)}&\ P_{\min}\ge-[C_t]\ (\text{nếu }[C_t]>0) &
\text{(H7)}&\ M^x_{\max},M^y_{\max}\le[M]\ (\text{nếu }[M]>0)
\end{aligned}
\end{equation}
với $s^\star$ là cặp gần nhất ($\min\{s_x,s_y\}$ kiểu A; $\min\{s_x,\sqrt{(s_x/2)^2+s_y^2}\}$ kiểu B).

Vi phạm gộp chuẩn hóa:
\begin{equation}
\Theta(\mathbf{x})=\sum_{r}\frac{\pos{g_r(\mathbf{x})}}{\rho_r}\ \ge 0,\qquad
\text{khả thi}\iff\Theta=0,\qquad
\Feas=\{\mathbf{x}:\Theta(\mathbf{x})=0\}.
\end{equation}

\subsection{Ràng buộc tùy chọn (mặc định TẮT; bật thì gộp vào $\Theta$)}
\begin{equation}
\text{(C1)}\ s^{\dagger}\le 6d;\qquad
\text{(C2)}\ H_{\max}\le[H];\qquad
\text{(C3)}\ \frac{P_{\max}}{[P_o]}+\frac{\max(M^x,M^y)}{[M]}\le1.
\end{equation}
C1 (cận trên $6d$) chủ yếu được áp như \emph{chặn biến} khi giải mã ($s_x,s_y\le6d$);
chỉ đường chéo kiểu B mới có thể vượt. C2,C3 (R7,R8) bật ở luồng mở rộng.

\subsection{Quan hệ trội có ràng buộc}
\begin{equation}
\mathbf{x}^{(a)}\prec\mathbf{x}^{(b)}\iff
\begin{cases}
\Theta_a=0\wedge\Theta_b>0,\\
0<\Theta_a<\Theta_b,\\
\Theta_a=\Theta_b=0\wedge (f_1,f_2)_a\preceq_{\mathrm{Pareto}}(f_1,f_2)_b.
\end{cases}
\end{equation}

\subsection{Hai tầng ngoài (mở rộng)}
\begin{equation}
d^\star=\arg\min_{d\in\Dset}\Big(n(d)\tfrac{\pi d^2}{4},\,n(d)\Big),\qquad
(L_x,L_y)\leftarrow\textsc{ResizeCap}(\mathbf{x}^\star).
\end{equation}

\section{PHẦN IV --- Thuật toán giải}
% =========================================================================
Vì $\Mcoc$ là hộp đen không khả vi, bài toán đa mục tiêu, biến hỗn hợp, miền khả thi
rất hẹp $\Rightarrow$ dùng \textbf{NSGA-II}: sắp xếp không-bị-trội nhanh + khoảng cách
chen chúc + SBX ($\eta_c{=}15$) + đột biến đa thức ($\eta_m{=}20$) + chọn lọc $\mu{+}\lambda$;
kèm \emph{gieo hạt tất định} (phủ vùng khả thi hẹp), \emph{cache} theo spec lưới và
\emph{trần ngân sách} số lần gọi $\Mcoc$. Pseudocode chi tiết: \texttt{docs/THUAT\_TOAN\_PSEUDOCODE.md}.
