\begin{abstract}\noindent
Bài toán được phát biểu theo quy ước tối ưu / vận trù học (OR): khối \emph{Tập hợp ---
Tham số --- Biến --- Hàm mục tiêu --- Ràng buộc --- Miền biến}, dạng một chương trình
quy hoạch phi tuyến hỗn hợp nguyên. Mọi ràng buộc là một thể hiện của bất đẳng thức
LRFD $\eta\,Q \le \phi\,R_n$ (TCVN 11823:2017 --- bản Việt hóa AASHTO LRFD); mọi trạng
thái giới hạn luôn áp dụng. Trị số $\phi,\gamma$ lấy theo AASHTO LRFD mà TCVN 11823 áp
dụng (đối chiếu bản chuẩn khi phát hành hồ sơ).
\end{abstract}

\section{Mô hình (P)}

\begin{equation*}
\textbf{(P)}\quad
\min_{t,n_x,n_y,s_x,s_y,D,L_c,B_x,B_y,h,A_s}\
Z = n\frac{\pi D^2}{4}L_c\,c_{bt} + n\,c_{tc} + B_xB_yh\,c_{bt} + W_s(A_s)\,c_s
\end{equation*}

\noindent\textbf{s.t.}
\begin{alignat}{2}
& \eta\,P_i^\ell \le \phi_c R_n^{\text{comp}}     && \quad \forall i\in\Iset,\ \forall\ell\in\{\text{Str,Ext}\} \tag{1}\\
& \eta\,T_i^\ell \le \phi_{up} R_n^{\text{upl}}    && \quad \forall i,\ \forall\ell\in\{\text{Str,Ext}\} \tag{2}\\
& \eta\,H_i^\ell \le \phi_\ell R_n^{\text{lat}}    && \quad \forall i,\ \forall\ell\in\{\text{Str,Ext}\} \tag{3}\\
& g^{PM}\!\big(P_i^\ell,M_i^\ell\big) \le 1         && \quad \forall i,\ \forall\ell\in\{\text{Str,Ext}\} \tag{4}\\
& \eta\,M_u^\ell \le \phi_f M_n(A_s,h,D)            && \quad \forall\ell\in\{\text{Str}\} \tag{5}\\
& \eta\,V_u^{1,\ell} \le \phi_v V_n^{\text{1way}}   && \quad \forall\ell\in\{\text{Str}\} \tag{6}\\
& \eta\,V_u^{2,\ell} \le \phi_v V_n^{\text{2way}}   && \quad \forall\ell\in\{\text{Str}\} \tag{7}\\
& \eta\,F_u^{stm,\ell} \le \phi_{stm} R_n^{\text{stm}} && \quad \forall\ell\in\{\text{Str}\}\ (\text{đài sâu}) \tag{8}\\
& S^\ell \le S_{tol}                                && \quad \forall\ell\in\{\text{Ser}\} \tag{9}\\
& \delta^\ell \le \delta_{tol}                      && \quad \forall\ell\in\{\text{Ser}\} \tag{10}\\
& \norm{\mathbf{p}_i-\mathbf{p}_j}\ge s_{\min}=\max(2.5D,0.75) && \quad \forall i,j\in\Iset,\ i<j \tag{11}\\
& \max_i|x_i|+\tfrac{D}{2}+e_{\min}\le\tfrac{B_x}{2};\ (\text{tương tự } y) && \tag{12}\\
& \{\mathbf{p}_i\}=\Gen(t,n_x,n_y,s_x,s_y) && \tag{13}\\
& \{P_i,H_i,M_i\}^\ell=\Anl(\cdot;Q^\ell),\ S^\ell=\Stl(\cdot),\ \delta^\ell=p\text{--}y && \quad \forall\ell\in\Lset \tag{14}\\
& t\in\{\mathrm A,\mathrm B\},\ n_x,n_y\in\Integer_+,\ s_x,s_y\in[s_{\min},6D], && \notag\\
& \quad D\in\Dset,\ L_c,B_x,B_y,h,A_s\in\Real_+ && \tag{15}
\end{alignat}

\section{Nhận xét (phân loại \& lời giải)}
\begin{itemize}
\item \textbf{(P) là quy hoạch phi tuyến hỗn hợp nguyên (MINLP)}: biến nguyên $(n_x,n_y)$,
phân loại $t$, rời rạc $D$, liên tục còn lại.
\item \textbf{Ràng buộc hộp đen}: toán tử phân tích $\Anl$ (MCOC) không có dạng đóng,
\emph{không khả vi, không lồi}; đa tổ hợp tải.
\item \textbf{Mọi ràng buộc đồng dạng LRFD} $\eta Q\le\phi R_n$ $\Rightarrow$ chuẩn hóa
$g_c(\mathbf{x})=\dfrac{\eta Q_c}{\phi_c R_{n,c}}-1\le0$, gộp thành độ đo vi phạm
$\Theta(\mathbf{x})=\sum_c\pos{g_c(\mathbf{x})}$; \emph{khả thi}$\iff\Theta=0$;
$\Feas=\{\mathbf{x}:\Theta(\mathbf{x})=0\}$.
\item \textbf{Lời giải}: không dùng được phương pháp gradient/đối ngẫu $\Rightarrow$
\emph{metaheuristic quần thể} (NSGA-II) với xử lý ràng buộc bằng \emph{trội có ràng buộc}
(constrained-domination) trên $\Theta$; oracle chính xác $=$ MCOC. Pseudocode:
\texttt{docs/THUAT\_TOAN\_PSEUDOCODE.md}; kế hoạch chuyển đổi: \texttt{plans/025-migrate-tcvn11823.md}.
\end{itemize}

\paragraph{Nguồn trị số.} $\gamma$ (Bảng 3.4.1-1), $\phi$ cọc (Đ.10.5.5.2.3), $\phi$ bê
tông (Đ.5.5.4.2), $s_{\min}$ \& mép (Đ.10.7.1.2), $p$-multiplier (Đ.10.7.2.4) --- theo
AASHTO LRFD mà TCVN 11823:2017 áp dụng; đối chiếu bản TCVN khi phát hành hồ sơ.
